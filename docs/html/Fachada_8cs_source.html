234 public List<Interaccion> VerInteraccionesDeCliente(int idCliente, string tipoDeInteraccion)
235 {
236 var cliente = _repoClientes.Buscar(idCliente);
237 if (cliente == null)
238 {
239 return new List<Interaccion>();
240 }
241
242 var interaccionesFiltradas = new List<Interaccion>();
243 string tipoBuscado = tipoDeInteraccion.ToLower(); // Normaliza a minúsculas
244
245 foreach (var interaccion in cliente.Interacciones)
246 {
247 bool coincide = false;
248 // Usa 'is' para chequear el tipo de objeto (Polimorfismo)
249 if (tipoBuscado == "llamada" && interaccion is Llamada)
250 {
251 coincide = true;
252 }
253 else if (tipoBuscado == "reunion" && interaccion is Reunion)
254 {
255 coincide = true;
256 }
257 else if (tipoBuscado == "mensaje" && interaccion is Mensaje)
258 {
259 coincide = true;
260 }
261 else if (tipoBuscado == "correo" && interaccion is Correo)
262 {
263 coincide = true;
264 }
265
266 if (coincide)
267 {
268 interaccionesFiltradas.Add(interaccion);
269 }
270 }
271
272 return interaccionesFiltradas;
273 }

525 public List<Cliente> ObtenerClientesInactivos(int diasSinInteraccion)
526 {
527 List<Cliente> clientesInactivos = new List<Cliente>();
528 // Calcula la fecha límite (ej: hoy - 30 días)
529 DateTime fechaLimite = DateTime.Now.AddDays(-diasSinInteraccion);
530
531 foreach (var cliente in _repoClientes.ObtenerTodos())
532 {
533 // Si no tiene interacciones, está inactivo.
534 if (cliente.Interacciones.Count == 0)
535 {
536 clientesInactivos.Add(cliente);
537 continue; // Salta al siguiente cliente
538 }
539
540 // Busca la fecha de la última interacción
541 DateTime fechaMasReciente = DateTime.MinValue;
542 foreach (var interaccion in cliente.Interacciones)
543 {
544 if (interaccion.Fecha > fechaMasReciente)
545 {
546 fechaMasReciente = interaccion.Fecha;
547 }
548 }
549
550 // Si la última interacción es más vieja que la fecha límite, está inactivo.
551 if (fechaMasReciente < fechaLimite)
552 {
553 clientesInactivos.Add(cliente);
554 }
555 }
556
557 return clientesInactivos;

575 public List<Cliente> ObtenerClientesSinRespuesta()
576 {
577 List<Cliente> clientesSinRespuesta = new List<Cliente>();
578
579 foreach (var cliente in _repoClientes.ObtenerTodos())
580 {
581 if (cliente.Interacciones.Count == 0)
582 {
583 continue; // Si no hay interacciones, no aplica.
584 }
585
586 // Busca la última interacción (la más reciente)
587 Interaccion ultimaInteraccion = null;
588 DateTime fechaMasReciente = DateTime.MinValue;
589
590 foreach (var interaccion in cliente.Interacciones)
591 {
592 if (interaccion.Fecha > fechaMasReciente)
593 {
594 fechaMasReciente = interaccion.Fecha;
595 ultimaInteraccion = interaccion;
596 }
597 }
598
599 // Si la última interacción no es nula y es una Llamada...
600 if (ultimaInteraccion != null && ultimaInteraccion is Llamada)
601 {
602 // 'as' convierte el tipo de forma segura (da null si no puede)
603 Llamada ultimaLlamada = ultimaInteraccion as Llamada;
604 if (ultimaLlamada.TipoLlamada == "Recibida")
605 {
606 clientesSinRespuesta.Add(cliente);
607 }
608 }
609 }
610
611 return clientesSinRespuesta;

620 public ResumenDashboard ObtenerResumenDashboard()
621 {
622 var todosLosClientes = _repoClientes.ObtenerTodos();
623 // 1. Total de clientes
624 int totalClientes = todosLosClientes.Count;
625
626 // 2. Busca todas las interacciones de todos los clientes
627 List<Interaccion> todasLasInteracciones = new List<Interaccion>();
628 foreach (Cliente cliente in todosLosClientes)
629 {
630 // Agrega todas las interacciones de este cliente a la lista grande
631 todasLasInteracciones.AddRange(cliente.Interacciones);
632 }
633
634 // Ordena de más nueva a más vieja (descendente)
635 todasLasInteracciones.Sort((i1, i2) => i2.Fecha.CompareTo(i1.Fecha));
636
637 // Se queda solo con las 5 primeras (las más recientes)
638 List<Interaccion> interaccionesRecientes = new List<Interaccion>();
639 int contadorMaximo = 5;
640
641 for (int i = 0; i < todasLasInteracciones.Count; i++)
642 {
643 if (i >= contadorMaximo)
644 {
645 break; // Para de agregar si ya tiene 5
646 }
647 interaccionesRecientes.Add(todasLasInteracciones[i]);
648 }
649
650 // 3. Busca reuniones que todavía no pasaron
651 List<Reunion> reunionesProximas = new List<Reunion>();
652 DateTime ahora = DateTime.Now;
653
654 foreach (Interaccion interaccion in todasLasInteracciones) // Reusa la lista total
655 {
656 if (interaccion is Reunion) // Si es una reunión
657 {
658 if (interaccion.Fecha > ahora) // Y es en el futuro
659 {
660 // Hacemos 'casting' explícito porque sabemos que es una Reunion
661 reunionesProximas.Add((Reunion)interaccion);
662 }
663 }
664 }
665
666 // Ordena las reuniones de más cercana a más lejana (ascendente)
667 reunionesProximas.Sort((r1, r2) => r1.Fecha.CompareTo(r2.Fecha));
668
669 // 4. Arma el objeto de resumen y lo devuelve
670 var resumen = new ResumenDashboard
671 {
672 TotalClientes = totalClientes,
673 InteraccionesRecientes = interaccionesRecientes,
674 ReunionesProximas = reunionesProximas
675 };
676
677 return resumen;
678 }
